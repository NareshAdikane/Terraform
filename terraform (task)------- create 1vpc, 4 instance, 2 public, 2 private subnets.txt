# VPC creation
resource "aws_vpc" "main" {
  cidr_block = var.vpc_cidr
  tags = {
    Name = "my-vpc"        # This name will appear on your vpc 
  }
}

# Public subnet
resource "aws_subnet" "public" {
  count                   = length(var.public_subnet_cidrs)
  cidr_block              = var.public_subnet_cidrs[count.index]
  vpc_id                  = aws_vpc.main.id        # this is the vpc id which is mention above with 
# mention subnets which are present in your AZ
  availability_zone       = element(["us-east-2a", "us-east-2b"], count.index)  
  map_public_ip_on_launch = true

  tags = {
    Name = "public-subnet-${count.index}"
  }
}

# Private subnet
resource "aws_subnet" "private" {
  count             = length(var.private_subnet_cidrs)
  cidr_block        = var.private_subnet_cidrs[count.index]
  vpc_id            = aws_vpc.main.id
  availability_zone = element(["us-east-2a", "us-east-2b"], count.index)

  tags = {
    Name = "private-subnet-${count.index}"
  }
}

# Instance Creation
resource "aws_instance" "my-instance" {
  count         = min(var.instance_count, length(aws_subnet.public) + length(aws_subnet.private))
  ami           = "ami-080c09858e04800a1"       # Give your AMI ID 
  instance_type = "t2.micro"              # give any instance type
  subnet_id     = count.index < length(aws_subnet.public) ? aws_subnet.public[count.index].id : aws_subnet.private[count.index - length(aws_subnet.public)].id

  tags = {
    Name = "my-instance-${count.index}"
  }
}


5. variable.tf :

# Defining vpc 
variable "vpc_cidr" {
  description = "The CIDR block for the VPC"
  type        = string
  default     = "10.0.0.0/16"
}

#Defining subnet (public)
variable "public_subnet_cidrs" {
  description = "List of CIDR blocks for public subnets"
  type        = list(string)
  default     = ["10.0.1.0/24", "10.0.2.0/24"]
}

# Defining subnet (private)
variable "private_subnet_cidrs" {
  description = "List of CIDR blocks for private subnets"
  type        = list(string)
  default     = ["10.0.3.0/24", "10.0.4.0/24"]
}

#Instance defining
variable "instance_count" {
  description = "Number of instances to launch"
  type        = number
  default     = 5      # add number of instances you want to launch
}
